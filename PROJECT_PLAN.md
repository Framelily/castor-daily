  const dayOfWeek = dayjs(date).day()

  // à¸§à¸±à¸™à¸«à¸¢à¸¸à¸” -> à¸ªà¸µà¹€à¸—à¸²à¸­à¹ˆà¸­à¸™
  if (dayOfWeek === profile.rest_day_index) {

  const completedTasks = tasks.filter(t => t.status === 'completed').length

  if (totalTasks === 0) return 'gray-100'

  const rate = completedTasks / totalTasks

  if (rate >= 0.7) return 'green-300'     // 70-99%

  return 'red-400'                         // < 40%
